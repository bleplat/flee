outfit Reinforced_Armor
	desc=Increase your ship's hull.
	sprite_coords=0;2
	sprite_color=UpgradeColor1
	required_stat=integrity >= 50
	time=150
	once_effect=ship.integrity *= 1.35
	effect=integrity += 15
	effect=integrity *= 1.30
	cost=250;0;0;0
